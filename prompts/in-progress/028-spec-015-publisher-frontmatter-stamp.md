In the `frontmatterFormatter.Format` impl, set the key AFTER the operator-merge loop and BEFORE the `created_by` force-set:

```go
func (f *frontmatterFormatter) Format(
	operator lib.TaskFrontmatter,
	slug string,
	date schedule.Date,
	autoAbortPrior bool,
) lib.TaskFrontmatter {
	out := lib.TaskFrontmatter{
		"status":    "in_progress",
		"page_type": "task",
	}
	for k, v := range operator {
		if s, ok := v.(string); ok {
			out[k] = f.renderer.Render(s, slug, date)
			continue
		}
		out[k] = v
	}
	out["auto_abort_prior"] = autoAbortPrior
	out["created_by"] = "recurring-task-creator"
	return out
}
```
